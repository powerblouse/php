<?php

    $n = $_GET["choice"];

    $length = strlen($n);

    $upper=0;

    $lower=0;

    for ($i=0;$i<$length;$i++)

    {

        $char = $n[$i];

        if(ctype_upper($char))

        {

            $upper+=1;

        }

        else{

            $lower+=1;

        }

    }

    echo "Uppercase letters: $upper";

    echo "Lowercase letters: $lower";
